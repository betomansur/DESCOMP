STA @290                    	#Zera o HEX 2

NOP                             #LOOP PRINCIPAL
LDA @353
AND @63                    	#Carrega o valor do KEY1 no acumulador

JEQ @39                       #Se 0, Linha de carregar o valor do key0
JSR @51           	        #Se 1, subrotina CONFIGURACAO LIMITE
LDA @352
AND @63                    	#Carrega o valor do KEY0 no acumulador
CEQ @62                     	#Compara o valor do KEY0 com 62 (0)
JEQ @45                       #Se 0, Linha de carregar o valor do reset
JSR @59                       #Se 1, SUBROTINA VERIFICACAO LIMITE
JSR @55                       #Se 1, subrotina INCREMENTO
LDA @356
AND @63                    	#Carrega o valor do Reset no acumulador
CEQ @62                     	#Compara o valor do Reset com 62 (0)
JEQ @51        	            #Se 0, linha de JMP inicial
JSR @63                       #Se 1, subrotina FPGA RESET
JMP @33          	            #Se nenhum dos botoes forem clicados, vai para o label Start (Laco principal)  linha ~34

RET
